142  std::map<colorIndex, std::unique_ptr<aiScene>> createNonPolyScenes() {
143  std::map<colorIndex, std::unique_ptr<aiScene>> scenes;
144 
145  std::map<colorIndex, std::vector<unsigned int>> colorsWithIndices;
146 
147  std::map<std::array<float, 3>, glm::vec3> summedVertexNormals;
148 
149  // key=edge, value=(tri_idx, vert_idx, vert_idx)
150  std::map<std::array<std::array<float, 3>, 2>, IndexedEdge> edgeLookup;
151 
152  for(unsigned int i = 0; i < mGeometry->getTriangleCount(); i++) {
153  colorIndex color = mGeometry->getTriangle(i).getColor();
154  colorsWithIndices[color].emplace_back(static_cast<unsigned int>(i));
155 
156  for(unsigned int j = 0; j < 3; j++) {
157  std::array<float, 3> vertex = {mGeometry->getTriangle(i).getVertex(j).x,
158  mGeometry->getTriangle(i).getVertex(j).y,
159  mGeometry->getTriangle(i).getVertex(j).z};
160 
161  summedVertexNormals[vertex] += mGeometry->getTriangle(i).getNormal();
162 
163  std::array<float, 3> nextVertex = {mGeometry->getTriangle(i).getVertex((j + 1) % 3).x,
164  mGeometry->getTriangle(i).getVertex((j + 1) % 3).y,
165  mGeometry->getTriangle(i).getVertex((j + 1) % 3).z};
166 
167  IndexedEdge &edge = edgeLookup[{vertex, nextVertex}];
168  edge.color = mGeometry->getTriangle(i).getColor();
169  edge.tri = i;
170  edge.id1 = j;
171  edge.id2 = (j + 1) % 3;
172  }
173  }
174 
175  normalizeSummedNormals(summedVertexNormals);
176 
177  computeBoundaryEdges(edgeLookup);
178 
179  for(auto &indexOfColor : colorsWithIndices) {
180  auto soloBoundary = selectBoundaryEdgesByColor(edgeLookup, indexOfColor.first);
181 
182  scenes[indexOfColor.first] = std::move(createNewNonPolyScene(
183  indexOfColor.second, summedVertexNormals, soloBoundary, mExtrusionCoef[indexOfColor.first]));
184  }
185 
186  return scenes;
187  }

197  void computeBoundaryEdges(std::map<std::array<std::array<float, 3>, 2>, IndexedEdge> &edgeLookup) {
198  for(auto &edge : edgeLookup) {
199  if(edgeLookup.count({edge.first[1], edge.first[0]})) {
200  if(!edge.second.isBoundary && edgeLookup[{edge.first[1], edge.first[0]}].color != edge.second.color) {
201  edgeLookup[{edge.first[1], edge.first[0]}].isBoundary = true;
202  edge.second.isBoundary = true;
203  }
204  }
205  }

223  std::map<colorIndex, std::unique_ptr<aiScene>> createPolyScenes(bool withSDF) {
224  std::map<colorIndex, std::unique_ptr<aiScene>> scenes;
225 
226  std::map<colorIndex, std::vector<DetailedTriangleId>> colorsWithIndices;
227 
228  std::unordered_map<PolyhedronData::vertex_descriptor, glm::vec3> summedVertexNormals;
229  std::map<PolyhedronData::vertex_descriptor, float> vertexSDF;
230 
231  std::map<colorIndex, std::set<PolyhedronData::halfedge_descriptor>> borderEdges;
232 
233  for(PolyhedronData::face_descriptor fd : mGeometry->getMeshDetailed()->faces()) {
234  colorIndex color = mGeometry->getTriangle(mGeometry->getMeshDetailedIdMap()[fd]).getColor();
235  colorsWithIndices[color].emplace_back(mGeometry->getMeshDetailedIdMap()[fd]);
236  }
237 
238  for(PolyhedronData::vertex_descriptor vd : mGeometry->getMeshDetailed()->vertices()) {
239  int degree = mGeometry->getMeshDetailed()->degree(vd);
240  std::vector<glm::vec3> vertexNormals;
241 
242  auto halfedge = mGeometry->getMeshDetailed()->halfedge(vd);
243 
244  for(size_t i = 0; i < degree; i++) {
245  auto face = mGeometry->getMeshDetailed()->face(halfedge);
246 
247  if(face.is_valid()) {
248  DetailedTriangleId triIndex = mGeometry->getMeshDetailedIdMap()[face];
249 
250  if(withSDF) {
251  vertexSDF[vd] += (float)mGeometry->getSdfValue(triIndex.getBaseId());
252  }
253 
254  vertexNormals.push_back(mGeometry->getTriangle(triIndex).getNormal());
255 
256  colorIndex faceColor = mGeometry->getTriangle(triIndex).getColor();
257 
258  auto oppositeHalfedge = mGeometry->getMeshDetailed()->opposite(halfedge);
259  auto oppositeFace = mGeometry->getMeshDetailed()->face(oppositeHalfedge);
260 
261  if(oppositeFace.is_valid()) {
262  DetailedTriangleId oppositeFaceIdx = mGeometry->getMeshDetailedIdMap()[oppositeFace];
263  colorIndex oppositeFaceColor = mGeometry->getTriangle(oppositeFaceIdx).getColor();
264  if(faceColor != oppositeFaceColor) {
265  borderEdges[faceColor].insert(halfedge);
266  }
267  } else {
268  // halfedge is border edge (hole in the model), so it is alsou boundary edge
269  borderEdges[faceColor].insert(halfedge);
270  }
271  }
272 
273  halfedge = mGeometry->getMeshDetailed()->next_around_target(halfedge);
274  }
275 
276  float ep = glm::epsilon<float>();
277 
278  bool isEpsSameNormal = false;
279  for(size_t i = 0; i < vertexNormals.size(); i++) {
280  for(size_t j = 0; j < i; j++) {
281  isEpsSameNormal |= glm::all(glm::epsilonEqual(vertexNormals[i], vertexNormals[j], ep));
282  }
283  if(!isEpsSameNormal) {
284  summedVertexNormals[vd] += vertexNormals[i];
285  }
286  isEpsSameNormal = false;
287  }
288 
289  summedVertexNormals[vd] = glm::normalize(summedVertexNormals[vd]);
290 
291  if(withSDF) {
292  vertexSDF[vd] = vertexSDF[vd] / degree; // average
293  }
294  }
295 
296  for(auto &indexOfColor : colorsWithIndices) {
297  scenes[indexOfColor.first] =
298  std::move(createNewPolyScene(indexOfColor.second, summedVertexNormals, borderEdges[indexOfColor.first],
299  vertexSDF, mExtrusionCoef[indexOfColor.first]));
300  }
301 
302  return scenes;
303  }

418  float extrusionCoef = glm::length(mGeometry->getBoundingBoxMax() - mGeometry->getBoundingBoxMin()) * userCoef;
419

423  scene->mMaterials = new aiMaterial *[1];

432 
433  scene->mMeshes[0] = new aiMesh();

436  scene->mRootNode->mMeshes = new unsigned int[1];
437  scene->mRootNode->mMeshes[0] = 0;

441 
442  size_t trianglesCount = triangleIndices.size();

454  face.mIndices = new unsigned int[3];
455  face.mNumIndices = 3;
456 
457  for(unsigned int j = 0; j < face.mNumIndices; j++) {

463  mGeometry->getTriangle(triangleIndices[i]).getNormal().y,
464  mGeometry->getTriangle(triangleIndices[i]).getNormal().z);

471  aiFace &face = pMesh->mFaces[i + trianglesCount];
472  face.mIndices = new unsigned int[3];
473  face.mNumIndices = 3;
474 
475  for(unsigned int j = 0; j < face.mNumIndices; j++) {
476  unsigned int jRevert = 2 - j;
477 
478  glm::vec3 vertex = mGeometry->getTriangle(triangleIndices[i]).getVertex(j);
479 
480  glm::vec3 vertexNormal = extrusionCoef * vertexNormalLookup[{vertex.x, vertex.y, vertex.z}];
481 
482  pMesh->mVertices[3 * (i + trianglesCount) + j] =
483  aiVector3D(vertex.x - vertexNormal.x, vertex.y - vertexNormal.y, vertex.z - vertexNormal.z);

488  -mGeometry->getTriangle(triangleIndices[i]).getNormal().z);

490  face.mIndices[jRevert] = (unsigned int)(3 * (i + trianglesCount) + j);
491  }
492  }

494  for(size_t i = 0; i < borderEdges.size(); i++) {
495  aiFace &face1 = pMesh->mFaces[(2 * trianglesCount) + (2 * i)];
496  face1.mIndices = new unsigned int[3];
497  face1.mNumIndices = 3;
498 
499  aiFace &face2 = pMesh->mFaces[(2 * trianglesCount) + (2 * i + 1)];
500  face2.mIndices = new unsigned int[3];
501  face2.mNumIndices = 3;

504  glm::vec3 vertex2 = mGeometry->getTriangle(borderEdges[i].tri).getVertex(borderEdges[i].id2);
505 
506  glm::vec3 vertexNormal1 = extrusionCoef * vertexNormalLookup[{vertex1.x, vertex1.y, vertex1.z}];
507  glm::vec3 vertexNormal2 = extrusionCoef * vertexNormalLookup[{vertex2.x, vertex2.y, vertex2.z}];
508 
509  pMesh->mVertices[3 * 2 * (trianglesCount + i) + 0] = aiVector3D(vertex1.x, vertex1.y, vertex1.z);
510  pMesh->mVertices[3 * 2 * (trianglesCount + i) + 1] =
511  aiVector3D(vertex1.x - vertexNormal1.x, vertex1.y - vertexNormal1.y, vertex1.z - vertexNormal1.z);
512  pMesh->mVertices[3 * 2 * (trianglesCount + i) + 2] = aiVector3D(vertex2.x, vertex2.y, vertex2.z);
513 
514  pMesh->mVertices[3 * 2 * (trianglesCount + i) + 3] = aiVector3D(vertex2.x, vertex2.y, vertex2.z);
515  pMesh->mVertices[3 * 2 * (trianglesCount + i) + 4] =
516  aiVector3D(vertex1.x - vertexNormal1.x, vertex1.y - vertexNormal1.y, vertex1.z - vertexNormal1.z);
517  pMesh->mVertices[3 * 2 * (trianglesCount + i) + 5] =
518  aiVector3D(vertex2.x - vertexNormal2.x, vertex2.y - vertexNormal2.y, vertex2.z - vertexNormal2.z);
519 
520  aiVector3D normal = calculateNormal({vertex2, vertex1, vertex1 - vertexNormal1});
521 
522  for(size_t j = 0; j < 6; j++) {
523  pMesh->mNormals[3 * 2 * (trianglesCount + i) + j] = normal;
524  }
525 
526  face1.mIndices[0] = (unsigned int)(3 * 2 * (trianglesCount + i) + 0);
527  face1.mIndices[1] = (unsigned int)(3 * 2 * (trianglesCount + i) + 1);
528  face1.mIndices[2] = (unsigned int)(3 * 2 * (trianglesCount + i) + 2);
529 
530  face2.mIndices[0] = (unsigned int)(3 * 2 * (trianglesCount + i) + 3);
531  face2.mIndices[1] = (unsigned int)(3 * 2 * (trianglesCount + i) + 4);
532  face2.mIndices[2] = (unsigned int)(3 * 2 * (trianglesCount + i) + 5);
533  }
534 
535  return scene;
