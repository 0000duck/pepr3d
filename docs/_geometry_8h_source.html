427  template <typename Object>
428  bool isTriangleInRadius(const Object& object, double radius, size_t triangleIdx) const {
429  P_ASSERT(triangleIdx < mTriangleBounds.size());
430  P_ASSERT(mTriangleBounds.size() == mTriangles.size());
431 
432  const Point3& sphereCenter = mTriangleBounds[triangleIdx].first;
433  const double sphereRadius = mTriangleBounds[triangleIdx].second;
434 
435  // Any sphere in contact with the cylinder is going to have this max squared distance
436  const double distanceLimitSquared = (radius + sphereRadius) * (radius + sphereRadius);
437 
438  if(CGAL::squared_distance(object, sphereCenter) <= distanceLimitSquared) {
439  return true;
440  } else {
441  return false;
442  }
443  }

593 template <typename StoppingCondition>
594 std::vector<size_t> Geometry::bucketSpread(const StoppingCondition& stopFunctor, std::deque<size_t>& toVisit,
595  std::unordered_set<size_t>& alreadyVisited) {
596  std::vector<size_t> trianglesToColor;
597  P_ASSERT(mPolyhedronData.indices.size() == mTriangles.size());
598 
599  while(!toVisit.empty()) {
600  // Remove yourself from queue and mark visited
601  const size_t currentVertex = toVisit.front();
602  toVisit.pop_front();
603  P_ASSERT(alreadyVisited.find(currentVertex) != alreadyVisited.end());
604  P_ASSERT(currentVertex < mTriangles.size());
605  P_ASSERT(toVisit.size() < mTriangles.size());
606 
607  // Catching because of unpredictable CGAL errors
608  try {
609  // Manage neighbours and grow the queue
610  addNeighboursToQueue(currentVertex, alreadyVisited, toVisit, stopFunctor);
611  } catch(CGAL::Assertion_exception& excp) {
612  CI_LOG_E("Exception caught. Returning immediately. " + excp.expression() + " " + excp.message());
613  throw std::runtime_error("Bucket spread failed inside the CGAL library.");
614  }
615 
616  // Add the triangle to the list
617  trianglesToColor.push_back(currentVertex);
618  }
619  return trianglesToColor;
620 }

708 template <typename StoppingCondition>
709 void Geometry::addNeighboursToQueue(const size_t currentVertex, std::unordered_set<size_t>& alreadyVisited,
710  std::deque<size_t>& toVisit, const StoppingCondition& stopFunctor) const {
711  const std::array<int, 3> neighbours = gatherNeighbours(currentVertex);
712  for(int i = 0; i < 3; ++i) {
713  if(neighbours[i] == -1) {
714  continue;
715  } else {
716  if(alreadyVisited.find(neighbours[i]) == alreadyVisited.end()) {
717  // New vertex -> visit it.
718  if(stopFunctor(neighbours[i], currentVertex)) {
719  toVisit.push_back(neighbours[i]);
720  alreadyVisited.insert(neighbours[i]);
721  }
722  }
723  }
724  }
725 }

729  std::unordered_set<DetailedTriangleId>& alreadyVisited,
730  std::deque<DetailedTriangleId>& toVisit,
731  const StoppingCondition& stopFunctor) const {
732  std::array<std::optional<DetailedTriangleId>, 3> neighbours = gatherNeighbours(currentVertex);

740  toVisit.push_back(*neighbours[i]);

742  }
743  }
744  }
