50  using Segment = CGAL::Segment_2<K>;
51 
52  std::vector<size_t> verticesToRemove;
53  const size_t vertexCount = poly.vertices_end() - poly.vertices_begin();
54  for(size_t i = 0; i < vertexCount; i++) {
55  auto currentVertex = std::next(poly.vertices_circulator(), i);
56 
57  const typename K::Line_2 neighbourLine(*std::prev(currentVertex), *std::next(currentVertex));

59  verticesToRemove.emplace_back(i);

73  static std::vector<DataTriangle::K::Point_3> pointsOnCircle(const DataTriangle::K::Circle_3& circle, int segments) {
74  using Point3 = DataTriangle::K::Point_3;
75  P_ASSERT(segments >= 3);
76  P_ASSERT(!circle.is_degenerate());
77 
78  auto plane = circle.supporting_plane();
79  const double radius = sqrt(CGAL::to_double(circle.squared_radius()));
80  const auto xBase = plane.base1() / CGAL::sqrt(CGAL::to_double(plane.base1().squared_length()));
81  const auto yBase = plane.base2() / CGAL::sqrt(CGAL::to_double(plane.base2().squared_length()));
82 
83  std::vector<DataTriangle::K::Point_3> points;
84 
85  for(size_t i = 0; i < segments; i++) {
86  const double circleCoord = (static_cast<double>(i) / segments) * 2 * glm::pi<double>();
87  const Point3 pt = circle.center() + xBase * cos(circleCoord) * radius + yBase * sin(circleCoord) * radius;
88 
89  points.push_back(pt);
90  }
91

98  static CGAL::Orientation shoelaceOrientationTest(const CGAL::Polygon_2<K>& poly) {
99  typename K::FT shoelaceSum = 0;
100 
101  const size_t vertexCount = poly.vertices_end() - poly.vertices_begin();
102  for(size_t i = 0; i < vertexCount; i++) {
103  auto currentVertexIt = poly.vertices_circulator() + i;
104  auto nextVertexIt = currentVertexIt + 1;
105 
106  shoelaceSum += currentVertexIt->x() * nextVertexIt->y();
107  shoelaceSum -= currentVertexIt->y() * nextVertexIt->x();
108  }
109 
110  if(shoelaceSum > 0)
111  return CGAL::Orientation::COUNTERCLOCKWISE;
112  if(shoelaceSum < 0)
113  return CGAL::Orientation::CLOCKWISE;
114 
115  return CGAL::Orientation::COLLINEAR;
116  }
