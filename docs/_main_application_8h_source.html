231  ci::fs::path getRequiredAssetPath(const ci::fs::path& relativePath) {
232  ci::fs::path path = getAssetPath(relativePath);
233  if(path.empty()) {
234  std::string message;

237  message += "\n\nThis file is necessary and since it was not found, Pepr3D has to be terminated. ";
238  message += "Downloading Pepr3D again or reinstalling it should solve this problem.";
239  pushDialog(Dialog(DialogType::FatalError, "Could not find a required asset", message, "Exit Pepr3D"));
240  throw AssetNotFoundException(
241  std::string("Could not find required asset ").append(relativePath.string()).c_str());
242  }
243  return path;
