466  public:
467  // Defined only for debug environments, to avoid accidentaly leaving this expensive check in
468 #if !defined(NDEBUG) || defined(PEPR3D_EDGE_CONSISTENCY_CHECK)
469  static bool isEdgeTraversable(const Point2& form, const Point2& to,
473  const std::map<size_t, PolygonSet>& coloredPolys) {
474  const Segment2 sharedEdge = form < to ? Segment2(form, to) : Segment2(to, form);
475 
476  // Map a lower sorted point of the edge to a higher sorted point of the edge
477  std::map<Point2, Point2> pointMap;
478 
479  for(auto& colorSetIt : coloredPolys) {
480  std::vector<PolygonWithHoles> polys(colorSetIt.second.number_of_polygons_with_holes());
481  colorSetIt.second.polygons_with_holes(polys.begin());
482 
483  for(PolygonWithHoles& polyWithHoles : polys) {
484  Polygon& poly = polyWithHoles.outer_boundary();
485  for(size_t vertIdx = 0; vertIdx < poly.size(); vertIdx++) {
486  auto vertIt = poly.vertices_circulator() + vertIdx;
487  auto nextVertIt = std::next(vertIt);
488 
489  // If this edge is on the edge of bounds add it to the map
490  if(sharedEdge.has_on(*vertIt) && sharedEdge.has_on(*nextVertIt) && *vertIt != *nextVertIt) {
491  if(*vertIt < *nextVertIt) {
492  if(pointMap.find(*vertIt) != pointMap.end()) {
493  return false;
494  }
495  pointMap[*vertIt] = *nextVertIt;
496  } else {
497  if(pointMap.find(*nextVertIt) != pointMap.end()) {
498  return false;
499  }
500  pointMap[*nextVertIt] = *vertIt;
501  }
502  }
503  }
504  }
505  }
506 
507  // Now test that the boundary edge is traversable with the gathered edges
508  Point2 currentPt = sharedEdge.min();
509  const Point2 endPt = sharedEdge.max();
510  while(currentPt != endPt) {
511  auto nextPtIt = pointMap.find(currentPt);
512  if(nextPtIt == pointMap.end()) {
513  return false;
514  }
515 
516  currentPt = nextPtIt->second;
517  }
518 
519  return true;
520  }
521 #endif
